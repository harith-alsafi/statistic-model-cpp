320  // Array that will store the values of sigma(xi),sigma(xi^2),
321  // sigma(xi^3)....sigma(xi^2n)
322  double X[2*degree+1];
323  for (i = 0; i < 2*degree+1; i++){
324  X[i]=0;
325  // consecutive positions of the array will store N,sigma(xi),
326  // sigma(xi^2),sigma(xi^3)....sigma(xi^2n)
327  for (j=0;j<n;j++){
328  X[i]=X[i]+pow(x[j],i);
329  }
330  }
331 
332  // B is the Normal matrix(augmented) that
333  // will store the equations, 'a' is for
334  // value of the final coefficients
335  double B[degree+1][degree+2],a[degree+1];
336  // Build the Normal matrix by storing the
337  // corresponding coefficients at the right
338  // positions except the last column of the matrix
339  for (i = 0; i <= degree; i++){
340  for (j=0;j<=degree;j++){
341  B[i][j]=X[i+j];
342  }
343  }
344 
345  //Array to store the values of sigma(yi),
346  // sigma(xi*yi),sigma(xi^2*yi)...sigma(xi^n*yi)
347  double Y[degree+1];
348  //consecutive positions will store sigma(yi)
349  // sigma(xi*yi),sigma(xi^2*yi)...sigma(xi^n*yi)
350  for (i = 0; i < degree+1; i++){
351  Y[i]=0;
352  for (j = 0; j < n; j++){
353  Y[i]=Y[i]+pow(x[j],i)*y[j];
354  }
355  }
356 
357  // load the values of Y as the last column of
358  // B(Normal Matrix but augmented)
359  for (i = 0; i <= degree; i++){
360  B[i][degree+1]=Y[i];
361  }
362 
363  //n is made n+1 because the Gaussian
364  // Elimination part below was for n equations,
365  // but here n is the degree of polynomial
366  // and for n degree we get n+1 equations
367  degree=degree+1;
368  //From now Gaussian Elimination starts(can be ignored)
369  // to solve the set of linear equations (Pivotisation)
370  for (i = 0; i < degree; i++){
371  for (k = i+1; k < degree; k++){
372  if (B[i][i]<B[k][i]){
373  for (j = 0; j <= degree; j++){
374  double temp=B[i][j];
375  B[i][j]=B[k][j];
376  B[k][j]=temp;
377  }
378  }
379  }
380  }
381 
382  //loop to perform the gauss elimination
383  for (i = 0; i < degree-1; i++) {
384  for (k= i+1; k < degree; k++){
385  double t=B[k][i]/B[i][i];
386  // make the elements below the pivot
387  // elements equal to zero or elimnate the variables
388  for (j = 0; j <= degree; j++){
389  B[k][j]=B[k][j]-t*B[i][j];
390  }
391  }
392  }

396  // make the variable to be calculated equal
