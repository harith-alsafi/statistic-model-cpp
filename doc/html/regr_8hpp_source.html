381  // Array that will store the values of sigma(xi),sigma(xi^2),
382  // sigma(xi^3)....sigma(xi^2n)
383  double X[2*degree+1];
384  for (i = 0; i < 2*degree+1; i++){
385  X[i]=0;
386  // consecutive positions of the array will store N,sigma(xi),
387  // sigma(xi^2),sigma(xi^3)....sigma(xi^2n)
388  for (j=0;j<n;j++){
389  X[i]=X[i]+pow(x[j],i);
390  }
391  }
392 
393  // B is the Normal matrix(augmented) that
394  // will store the equations, 'a' is for
395  // value of the final coefficients
396  double B[degree+1][degree+2],a[degree+1];
397  // Build the Normal matrix by storing the
398  // corresponding coefficients at the right
399  // positions except the last column of the matrix
400  for (i = 0; i <= degree; i++){
401  for (j=0;j<=degree;j++){
402  B[i][j]=X[i+j];
403  }
404  }
405 
406  //Array to store the values of sigma(yi),
407  // sigma(xi*yi),sigma(xi^2*yi)...sigma(xi^n*yi)
408  double Y[degree+1];
409  //consecutive positions will store sigma(yi)
410  // sigma(xi*yi),sigma(xi^2*yi)...sigma(xi^n*yi)
411  for (i = 0; i < degree+1; i++){
412  Y[i]=0;
413  for (j = 0; j < n; j++){
414  Y[i]=Y[i]+pow(x[j],i)*y[j];
415  }
416  }
417 
418  // load the values of Y as the last column of
419  // B(Normal Matrix but augmented)
420  for (i = 0; i <= degree; i++){
421  B[i][degree+1]=Y[i];
422  }
423 
424  //n is made n+1 because the Gaussian
425  // Elimination part below was for n equations,
426  // but here n is the degree of polynomial
427  // and for n degree we get n+1 equations
428  degree=degree+1;
429  //From now Gaussian Elimination starts(can be ignored)
430  // to solve the set of linear equations (Pivotisation)
431  for (i = 0; i < degree; i++){
432  for (k = i+1; k < degree; k++){
433  if (B[i][i]<B[k][i]){
434  for (j = 0; j <= degree; j++){
435  double temp=B[i][j];
436  B[i][j]=B[k][j];
437  B[k][j]=temp;
438  }
439  }
440  }
441  }

443  //loop to perform the gauss elimination
444  for (i = 0; i < degree-1; i++) {
445  for (k= i+1; k < degree; k++){
446  double t=B[k][i]/B[i][i];
447  // make the elements below the pivot
448  // elements equal to zero or elimnate the variables
449  for (j = 0; j <= degree; j++){

457  // make the variable to be calculated equal
