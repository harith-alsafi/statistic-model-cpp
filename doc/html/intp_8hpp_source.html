46  virtual long double _find_value(long double xx,
47  std::vector<long double> _x, std::vector<long double> _y){
48  // converts x and y to std::pair so we can sort it without disrupting layout of points
49  std::vector<std::pair<long double, long double>> _points;
50  for(int i = 0; i < _x.size(); i++){
51  _points.push_back({_x[i], _y[i]});
52  }
53  std::sort(_points.begin(), _points.end());
54 
55  //point pair is < xx
56  auto lessThan =
57  [](std::pair<double, double> point, double xxx)
58  {return point.first < xxx;};
59 
60  //Find the first value >= xx
61  auto iter =
62  std::lower_bound(_points.cbegin(), _points.cend(), xx, lessThan);
63 
64  //xx > than the largest val in the x, we can't interpolate.
65  if(iter == _points.cend()) {
66  return (_points.cend() - 1)->second;
67  }
68 
69  //xx < than the smallest val in the x, we can't interpolate.
70  if(iter == _points.cbegin() && xx <= _points.cbegin()->first) {
71  return _points.cbegin()->second;
72  }
73 
74  // interpolating
75  long double upperX = iter->first;
76  long double upperY = iter->second;
77  long double lowerX = (iter - 1)->first;
78  long double lowerY = (iter - 1)->second;
79  long double deltaY = upperY - lowerY;
80  long double deltaX = upperX - lowerX;
81  long double yy = lowerY + ((xx - lowerX)/ deltaX) * deltaY;
82 
83  x_comb.push_back(xx);
84  y_comb.push_back(yy);
85  x_in.push_back(xx);
86  y_in.push_back(yy);
87  return yy;
88  }
