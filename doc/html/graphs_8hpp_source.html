115 string wrap(const char *const str, const size_t line_length)
116 {
117  char words[strlen(str) + 1];
118  strcpy(words, str);
119  string wrapped;
120 
121  size_t index = 0;
122  size_t linelen = 0;
123 
124  while (words[index] != '\0')
125  {
126  if (words[index] == '\n')
127  {
128  linelen = 0;
129  }
130  else if (isspace(words[index]))
131  {
132  size_t tempindex = index + 1;
133  size_t templinelen = linelen;
134  while (!isspace(words[tempindex]) and words[tempindex] != '\0')
135  {
136  ++templinelen;
137 
138  ++tempindex;
139  }
140 
141  char temp[templinelen + 1];
142  strncpy(temp, words + (index - linelen), templinelen);
143  temp[templinelen] = '\0';
144 
145  size_t width = strcol(temp);
146 
147  if (width >= line_length)
148  {
149  words[index] = '\n';
150  linelen = 0;
151  }
152  }
153 
154  if (words[index] == '\t')
155  linelen += 8 - (linelen % 8);
156  else if (words[index] != '\n')
157  ++linelen;
158 
159  ++index;
160  }
161  wrapped = words;
162  return wrapped;
